env setup
```
# install pyenv if you don't have it already
brew update && brew install pyenv
pyenv install 3.7.0

# set global default version of python
pyenv global 3.7.0

# Add for zsh set pyenv path for all sessions
if command -v pyenv 1>/dev/null 2>&1; then
  eval "$(pyenv init -)"
fi
```
